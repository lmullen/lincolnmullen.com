File loading madness:

file1 <- read.csv("file1.csv")
file2 <- read.csv("file2.csv")
# ...
file999 <- read.csv("file999.csv")

A better way:

# Old
tract01 <- scan("tract01.txt", what = "character", sep = "\n")

# New
file_names <- c("tract01.txt", "tract02.txt", "tract03.txt")
tracts <- lapply(file_names, scan, what = "character", sep = "\n")

str(tracts)
tracts[[1]]

Even better:

file_names <- Sys.glob("tract*.txt")
tracts <- lapply(file_names, scan, what = "character", sep = "\n")

str(tracts)
tracts[[1]]

Use lapply() and sapply()

lapply(tracts, length)
sapply(tracts, length)
